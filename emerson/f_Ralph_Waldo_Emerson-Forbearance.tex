\articletitle{Forbearance}

\articleauthor{Ralph Waldo Emerson}
\begin{poem}
	\begin{stanza}
		Hast thou named all the birds without a gun?\verseline
		Loved the wood-rose, and left it on its stalk?\verseline
		At rich men's tables eaten bread and pulse?\verseline
		Unarmed, faced danger with a heart of trust?\verseline
		And loved so well a high behavior,\verseline
		In man or maid, that thou from speech refrained,\verseline
		Nobility more nobly to repay?\verseline
		O, be my friend, and teach me to be thine!\verseline
	\end{stanza}
\end{poem}
